% Макрос для вставки рисунка по центру.
% Первый параметр (необязательный) - ширина рисунка.
% Второй параметр - путь до рисунка.
% Третий параметр - подпись к рисунку.
\newcommand{\melimg}[3][\textwidth]
{
    \begin{figure}[H]
        \center{\includegraphics[scale=1,width=#1]{#2}}
        \caption{#3}
        \label{img:#2}
    \end{figure}
}

% Для приложения.
\newcommand{\melappendix}[1]
{
    \newpage
    \stepcounter{section}
    \section*{\hfillПриложение \Asbuk{section}}
    \begin{center}
        #1
    \end{center}
    \markboth{#1}{}
    \addcontentsline{toc}{section}{Приложение \Asbuk{section}. #1}
}
